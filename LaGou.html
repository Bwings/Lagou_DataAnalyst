def cut_word(word,method):
    position = word.find('_')
    length = len(word)
    if position !=-1:
        bottomSalray = word[:position-1]
        topSalray = word[position+1:length-1]
    else:
        bottomSalary = word[:word.upper().find('K')]
        topSalary = bottomSalary
    if method == 'bottom' :
        return bottomSalary
    else:
        return topSalary
    #return bottomSalary,topSalary
